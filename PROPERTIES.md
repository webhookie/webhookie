##All possible properties

```
WEBHOOKIE_ANALYTICS_SEND
WH_ANALYTICS_SEND: boolean # default = true

WEBHOOKIE_CONSUMER_QUEUE
WH_CONSUMER_QUEUE: string #default = wh-consumer.event

WEBHOOKIE_CONSUMER_ADD_DEFAULT_GROUP
WH_CONSUMER_ADD_DEFAULT_GROUP: boolean # default = true

WEBHOOKIE_CONSUMER_MISSING_HEADER_EXCHANGE
WH_CONSUMER_MISSING_HEADER_EXCHANGE: string # default = wh-consumer

WEBHOOKIE_CONSUMER_MISSING_HEADER_ROUTING_KEY
WH_CONSUMER_MISSING_HEADER_ROUTING_KEY: string # default = wh-missing-header



WEBHOOKIE_SUBSCRIPTION_RETRY_MAX_RETRY
WH_SUBSCRIPTION_RETRY_MAX: integer # default = 3

WEBHOOKIE_SUBSCRIPTION_RETRY_INITIAL_INTERVAL
WH_SUBSCRIPTION_RETRY_INITIAL_INTERVAL: integer # default = 10

WEBHOOKIE_SUBSCRIPTION_RETRY_MULTIPLIER
WH_SUBSCRIPTION_RETRY_MULTIPLIER: integer # default = 2



WEBHOOKIE_SECURITY_AUDIENCE
WH_SECURITY_AUD: string # default = account

WEBHOOKIE_SECURITY_CLIENT_ID
WH_SECURITY_CLIENT_ID: string # default = webhookie-client

WEBHOOKIE_SECURITY_GROUPS_JWK_JSON_PATH
WH_SECURITY_GROUPS_JSON_PATH: jsonPath # default = $$.groups

WEBHOOKIE_SECURITY_OAUTH2_AUTHORIZATION_URI
WH_SECURITY_OAUTH2_AUTHORIZATION_URI: uri # default = /protocol/openid-connect/auth 

WEBHOOKIE_SECURITY_OAUTH2_TOKEN_URI
WH_SECURITY_OAUTH2_TOKEN_URI: uri # default = /protocol/openid-connect/token

WEBHOOKIE_SECURITY_OAUTH2_TOKEN_NAME
WH_SECURITY_OAUTH2_TOKEN_NAME: string # default = access_token

WEBHOOKIE_SECURITY_ENTITY_JWK_JSON_PATH
WH_SECURITY_ENTITY_JSON_PATH: jsonPath # default = $$.entity

WEBHOOKIE_SECURITY_ROLES_JWK_JSON_PATH
WH_SECURITY_ROLES_JSON_PATH: jsonPath # default = $$.realm_access.roles

WEBHOOKIE_SECURITY_ROLES_AUTO_ASSIGN_CONSUMER
WH_SECURITY_AUTO_ASSIGN_CONSUMER_ROLE: boolean # default = true

WEBHOOKIE_SECURITY_ALLOWED_ORIGINS: string[], # e.g http://localhost:4200, http://localhost:4300




```
